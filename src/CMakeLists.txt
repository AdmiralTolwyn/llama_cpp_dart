cmake_minimum_required(VERSION 3.22.1)

project(llama_cpp_dart_wrapper)

# ─── Android pthread shim ──────────────────────────────────────────
# Android's bionic libc has pthread baked in — there is no separate
# `-lpthread` library to link against, AND CMake's stock FindThreads
# probe calls `pthread_cancel`, which bionic does not expose. Both
# make the default `find_package(Threads REQUIRED)` flow fail at
# configure time.
#
# The cache-variable approach doesn't work: FindThreads.cmake
# unsets CMAKE_HAVE_LIBC_PTHREAD before its CHECK_C_SOURCE_COMPILES
# probe, so any pre-populated value is wiped. The canonical fix
# (used by the Android NDK team and many cross-compile templates)
# is to drop a tiny stub FindThreads.cmake into the build tree and
# prepend its directory to CMAKE_MODULE_PATH so it overrides the
# stock module for every downstream find_package(Threads) call —
# our wrapper, llama.cpp, ggml, mtmd, and any transitive consumers.
#
# The stub returns Threads::Threads as an empty INTERFACE imported
# target. That's correct on Android because pthread is provided by
# libc and needs no extra link flag.
#
# iOS / macOS / Linux / Windows are unaffected — the override is
# strictly gated on if(ANDROID).
if (ANDROID)
    set(_LCD_FT_STUB_DIR "${CMAKE_BINARY_DIR}/llama_cpp_dart_cmake_overrides")
    file(MAKE_DIRECTORY "${_LCD_FT_STUB_DIR}")
    file(WRITE "${_LCD_FT_STUB_DIR}/FindThreads.cmake" "\
# Auto-generated by llama_cpp_dart for Android.
# Replaces CMake's stock FindThreads which fails on bionic libc
# (no -lpthread, no pthread_cancel). Threads come from libc on
# Android — link flag is empty, the imported target is interface-only.
set(Threads_FOUND TRUE)
set(THREADS_FOUND TRUE)
set(CMAKE_THREAD_LIBS_INIT \"\")
set(CMAKE_USE_PTHREADS_INIT 1)
set(CMAKE_HAVE_LIBC_PTHREAD 1)
set(THREADS_PREFER_PTHREAD_FLAG ON)
if(NOT TARGET Threads::Threads)
  add_library(Threads::Threads INTERFACE IMPORTED)
endif()
")
    list(PREPEND CMAKE_MODULE_PATH "${_LCD_FT_STUB_DIR}")
endif()

# ─── llama.cpp source bootstrap ────────────────────────────────────
# When this plugin is consumed via `dart pub`, the Dart pub client
# clones the plugin's git repo WITHOUT recursing submodules (Dart's
# git package source does not support submodule init). That leaves
# `src/llama.cpp/` empty in the user's pub-cache, and the
# `add_subdirectory(llama.cpp)` call below explodes.
#
# When the plugin is consumed as a path dependency (or built from a
# normal `git clone --recurse-submodules`) the folder is populated
# and this block is a fast no-op.
#
# The bootstrap pins to the same commit recorded in `.gitmodules` so
# pub-cache builds match the upstream submodule tip. Don't edit the
# pin here directly — run `./update_submodules.sh` at the repo root
# and copy the resolved SHA in.
set(_LCD_LLAMA_CPP_DIR "${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp")
set(_LCD_LLAMA_CPP_REPO "https://github.com/ggml-org/llama.cpp.git")
set(_LCD_LLAMA_CPP_PIN  "15fa3c493bfcd040b5f4dcb29e1c998a0846de16")
if (NOT EXISTS "${_LCD_LLAMA_CPP_DIR}/CMakeLists.txt")
    message(STATUS "llama.cpp source missing; cloning into ${_LCD_LLAMA_CPP_DIR}")
    find_package(Git QUIET)
    if (NOT Git_FOUND)
        message(FATAL_ERROR
            "git is required to bootstrap llama.cpp into the plugin "
            "(pub-cache installs do not include submodules). Install "
            "git or build the plugin with the submodule pre-populated.")
    endif()
    file(MAKE_DIRECTORY "${_LCD_LLAMA_CPP_DIR}")
    execute_process(
        COMMAND ${GIT_EXECUTABLE} clone --depth 1 --no-tags
                "${_LCD_LLAMA_CPP_REPO}" "${_LCD_LLAMA_CPP_DIR}"
        RESULT_VARIABLE _lcd_clone_rc)
    if (NOT _lcd_clone_rc EQUAL 0)
        message(FATAL_ERROR "Failed to clone llama.cpp (rc=${_lcd_clone_rc})")
    endif()
    execute_process(
        COMMAND ${GIT_EXECUTABLE} -C "${_LCD_LLAMA_CPP_DIR}" fetch --depth 1 origin "${_LCD_LLAMA_CPP_PIN}"
        RESULT_VARIABLE _lcd_fetch_rc)
    if (_lcd_fetch_rc EQUAL 0)
        execute_process(
            COMMAND ${GIT_EXECUTABLE} -C "${_LCD_LLAMA_CPP_DIR}" checkout --detach FETCH_HEAD
            RESULT_VARIABLE _lcd_co_rc)
        if (NOT _lcd_co_rc EQUAL 0)
            message(WARNING
                "Failed to check out pinned llama.cpp commit "
                "${_LCD_LLAMA_CPP_PIN} (rc=${_lcd_co_rc}); using whatever "
                "the default branch points to instead.")
        endif()
    else()
        message(WARNING
            "Failed to fetch pinned llama.cpp commit ${_LCD_LLAMA_CPP_PIN} "
            "(rc=${_lcd_fetch_rc}); using whatever the default branch "
            "points to instead.")
    endif()
    # Recurse into llama.cpp's own submodules (kompute, etc.).
    execute_process(
        COMMAND ${GIT_EXECUTABLE} -C "${_LCD_LLAMA_CPP_DIR}" submodule update --init --recursive
        RESULT_VARIABLE _lcd_sm_rc)
    if (NOT _lcd_sm_rc EQUAL 0)
        message(WARNING "llama.cpp submodule update failed (rc=${_lcd_sm_rc})")
    endif()
endif()

# 1. Setup OpenCL paths BEFORE adding llama.cpp
set(OPENCL_HEADERS_DIR "${CMAKE_CURRENT_SOURCE_DIR}/OpenCL-Headers")

if (ANDROID)
    message(STATUS "Configuring for Android (OpenCL)...")
    message(STATUS "CMAKE_ANDROID_ARCH_ABI = ${CMAKE_ANDROID_ARCH_ABI}")

    # Define the path to the prebuilt library
    set(OPENCL_LIB_PATH "${CMAKE_CURRENT_SOURCE_DIR}/opencl-libs/android/${CMAKE_ANDROID_ARCH_ABI}/libOpenCL.so")

    # Create the imported target for your wrapper to use later
    add_library(opencl-lib SHARED IMPORTED)
    set_target_properties(opencl-lib PROPERTIES IMPORTED_LOCATION "${OPENCL_LIB_PATH}")
    
    # CRITICAL: Force llama.cpp (and ggml) to use these paths instead of searching system paths
    set(OpenCL_FOUND TRUE CACHE BOOL "OpenCL found" FORCE)
    set(OpenCL_INCLUDE_DIR "${OPENCL_HEADERS_DIR}" CACHE PATH "OpenCL include dir" FORCE)
    set(OpenCL_LIBRARY "${OPENCL_LIB_PATH}" CACHE FILEPATH "OpenCL library" FORCE)
endif()

# 2. Add llama.cpp (This builds llama, ggml, and applies the -DGGML_OPENCL flags)
# Hexagon NPU support (Qualcomm SM8450+ / Snapdragon 8 Gen 1+)
# Enable with: -DGGML_HEXAGON=ON in your Android build
# Requires libcdsprpc.so on the device (add <uses-native-library> to manifest)
add_subdirectory(llama.cpp)

# 3. Define your wrapper library (mtmd)
if (ANDROID)
    find_library(log-lib log)

    # Mirror upstream llama.cpp's `tools/mtmd/CMakeLists.txt` source
    # list. Newer llama.cpp split clip graphs out into per-model
    # files under `tools/mtmd/models/`; missing any of them produces
    # `undefined symbol: vtable for clip_graph_*` link errors. The
    # glob keeps us forward-compatible when new models are added —
    # it must be re-evaluated on every CMake re-configure (we set
    # CONFIGURE_DEPENDS) so a fresh `flutter pub get` picks them up.
    file(GLOB _LCD_MTMD_MODEL_SRCS CONFIGURE_DEPENDS
        "${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp/tools/mtmd/models/*.cpp")
    add_library(mtmd SHARED
        llama.cpp/tools/mtmd/mtmd.cpp
        llama.cpp/tools/mtmd/mtmd-audio.cpp
        llama.cpp/tools/mtmd/mtmd-image.cpp
        llama.cpp/tools/mtmd/clip.cpp
        llama.cpp/tools/mtmd/mtmd-helper.cpp
        ${_LCD_MTMD_MODEL_SRCS}
    )

    # Include directories
    target_include_directories(mtmd PUBLIC
        "${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp/include"
        "${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp/ggml/include"
        "${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp/common"
        "${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp/tools/mtmd"
        "${CMAKE_CURRENT_SOURCE_DIR}"
        "${OPENCL_HEADERS_DIR}"
    )

    # Link libraries
    target_link_libraries(mtmd 
        PUBLIC 
            llama 
            ggml
            ${log-lib}
            opencl-lib 
    )

    find_package(Threads REQUIRED)
    target_link_libraries(mtmd PRIVATE Threads::Threads)
endif()
